import datetime
today_date = datetime.datetime.today().date()
week_start = today_date - datetime.timedelta(days=today_date.weekday() + 1)
for i in range(24):
    pre_week_start = week_start - datetime.timedelta(weeks=i)
    pre_week_end = pre_week_start + datetime.timedelta(days=6)
    if not i:
        pre_week_end = today_date
    print(pre_week_start, pre_week_end)
